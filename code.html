local lib = loadstring(game:HttpGet("https://raw.githubusercontent.com/bbbPriv/BOOGA-UI/main/booga%20ui.lua"))()
local title = "Auto Killer"

for _, child in game:GetService("CoreGui"):GetChildren() do
    if child.Name == title and child:IsA("ScreenGui") then
        child:Destroy()
    end
end

-- JSON saving / loading
local Folder = "folder"
local Name = Folder.."/config.json"

settings = {}

if not isfolder(folder) then
    makefolder(folder)
    writefile(Name, game:service'HttpService':JSONEncode(settings))
end

function Load() 
    pcall(function()
        settings = game:service'HttpService':JSONDecode(readfile(Name))
    end)
end

function Write() 
    writefile(Name, game:service'HttpService':JSONEncode(settings))
end
