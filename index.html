Library Documentation

Your guide to using the Library effectively

Library.New(title)

Title: <string> - Title of the UI
Return: Library <table> 
local myLibrary = Library.New("My Awesome UI")

Library:Toggle()

Return: nil - Toggles UI 
myLibrary:Toggle()

Library:Notify(Title, Text, Position, Direction, Callback)

Title: <string> - Title of notification
Text: <string> - Description of notification
Position: <UDim2> - Position of start of the notification
Direction: <UDim2> - Direction of the notification
Callback: <function>
Return: (value)
Value: <boolean> - Accepted or cancelled 
myLibrary:Notify("Notice", "This is a notification", UDim2.new(0,0,0,0), UDim2.new(1,0,1,0), function(value)
    print("Notification result:", value)
end)

Library:AddPage(Title, Icon)

Title: <string> - Title of page
Icon: <string> - Asset ID of icon (optional)
Return: Pages <table> 
local myPage = myLibrary:AddPage("Home", "rbxassetid://123456")

Pages:AddSection(Title)

Title: <string> - Title of section
Return: Sections <table> 
local mySection = myPage:AddSection("Main Section")

Pages:AddSearchBar()

Return: nil - Adds Search Bar to the page 
myPage:AddSearchBar()

Sections:AddButton(Title, Callback)

Title: <string> - Title of button
Callback: <function>
Return: button <Instance> 
mySection:AddButton("Click Me", function()
    print("Button clicked")
end)

Sections:AddToggle(Title, IsEnabled, Callback)

Title: <string> - Title of toggle
IsEnabled: <boolean> - Default value of toggle
Callback: <function>
Return: (value, update)
Value: <boolean> - New toggle value
Return: toggle <Instance> 
mySection:AddToggle("Enable Feature", true, function(value)
    print("Toggle value:", value)
end)

Sections:AddTextBox(Title, Callback)

Title: <string> - Title of textbox
Callback: <function>
Return: (text, focus)
Text: <string> - Textbox content
Focus: <boolean> - Focus lost signal
Return: textbox <Instance> 
mySection:AddTextBox("Enter Text", function(text)
    print("Text entered:", text)
end)

Sections:AddKeybind(Title, Key, Callback)

Title: <string> - Title of keybind
Key: <keycode> - Default binded key
Callback: <function>
Return: (update)
Key: <KeyCode> - Key pressed
Return: keybind <Instance> 
mySection:AddKeybind("Bind Key", Enum.KeyCode.F, function()
    print("Key pressed")
end)

Sections:AddSlider(Title, Value, Min, Max, FixValues, Callback)

Title: <string> - Title of slider
Value: <number> - Default value of slider
Min: <number> - Minimum value of slider
Max: <number> - Maximum value of slider
FixValues: <number> - Fix values
Callback: <function>
Return: (value)
Value: <number> - New slider value
Return: slider <Instance> 
mySection:AddSlider("Adjust Value", 50, 0, 100, 1, function(value)
    print("Slider value:", value)
end)
